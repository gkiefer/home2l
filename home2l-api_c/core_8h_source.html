198  TCCR0A = 0; // Normal port operation (OC1A/OC1B disconnected); No waveform generation

214 static inline void MinitimerStart () {
215  TCCR1A = 0; // Normal port operation (OC1A/OC1B disconnected); No waveform generation
216  TCCR1B = (1 << PSR1) | 4; // Prescaler reset, set prescaler to CK/8
217  TC1H = 0; // Set high byte to 0 (affects TCNT1, OCR1C and any other 10 bit registers)
218  OCR1C = 0xff; // "top" value for counter
219  PLLCSR = 0; // disable PLL, ensure synchronous mode
220 }
221 
222 static inline void MinitimerStop () {
223  TCCR1B = 0;
224 }
225 
226 #define MinitimerReset() do { TCNT1 = 0; TCCR1B |= (1 << PSR1); } while (0)
227  // Note: According to the manual, "Due to synchronization of the CPU, Timer/Counter1
228  // data written into Timer/Counter1 is delayed by one and half CPU clock cycles
229  // in synchronous mode". Hence, we add a delay here before reading TCNT1.
